function calculateArea(width, height) {
  if (width < 0 || height < 0) {
    return "Invalid dimensions";
  }
  let area = width + height; // Error is here
  return area;
}

Hint: The area of a rectangle is calculated by multiplying width and height, not adding them.

Solution: Change let area = width + height; to let area = width * height;
